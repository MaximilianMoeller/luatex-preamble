% Provides: the ‘babel’ package, setting the language to english
% Requires: /

% Mutually exclusive with the ‘02-language-german’ template
% Will throw an error if both are loaded.

\iftoggle{02-language-german}{
	\PackageError{preamble/language}{%
		Only one language-header should be loaded! Previously loaded: german.
	}{
		Make sure to only load one language-header in the preamble of your document.
	}
}{
	\usepackage[english]{babel}
	\toggletrue{02-language-english}
}
